def display_cross_per_day_metrics(cross_results_dict, target_report_labels):
    """
    Displays per-day metrics for cross-dataset evaluation.

    Args:
        cross_results_dict (dict): The dictionary containing cross-evaluation results.
        target_report_labels (list): List of attack class names to display.
    """
    print("\n--- Cross-Dataset Per-Day Metrics ---")
    for scenario_key, scenario_results in cross_results_dict.items():
        print(f"\n################################################################################")
        print(f"### Scenario: {scenario_key} ###")
        print(f"################################################################################")

        if isinstance(scenario_results, str) or "Error" in scenario_results:
            print(f"  {scenario_results}")
            continue

        for model_name, model_data in scenario_results.items():
            print(f"\n--- Model: {model_name} ---")
            
            # Overall metrics for the combined test set in this scenario
            overall_metrics = model_data.get('Overall Metrics (Combined Test Set)', {})
            print("\n  --- Overall Metrics for Combined Test Set ---")
            print(f"    Accuracy: {overall_metrics.get('Accuracy', 'N/A'):.4f}")
            print(f"    F1-Score (weighted): {overall_metrics.get('F1-Score', 'N/A'):.4f}")
            print(f"    F1-Score (weighted, Attack Only): {overall_metrics.get('F1-Score (Attack Only)', 'N/A'):.4f}")
            print(f"    Recall (weighted): {overall_metrics.get('Recall', 'N/A'):.4f}")
            # Add other overall metrics as desired

            # Per-Day Metrics for the test set in this scenario
            print("\n  --- Per-Day Metrics for Test Set ---")
            per_day_metrics = model_data.get('Per_Day_Metrics', {})

            if not per_day_metrics:
                print("    No individual daily test results available for this scenario.")
                continue

            for day_name, day_results in per_day_metrics.items():
                print(f"\n    Day: {day_name}")
                if isinstance(day_results, str) or "Error" in day_results:
                    print(f"      {day_results}")
                    continue

                specific_day_metrics = day_results.get('Specific Attack Metrics', {})
                data = {}
                for target_label in target_report_labels:
                    if target_label in specific_day_metrics:
                        data[target_label] = {
                            'Precision': specific_day_metrics[target_label]['Precision'],
                            'Recall': specific_day_metrics[target_label]['Recall'],
                            'F1-Score': specific_day_metrics[target_label]['F1-Score'],
                            'Support': specific_day_metrics[target_label]['Support']
                        }
                    else:
                        data[target_label] = {
                            'Precision': np.nan, 'Recall': np.nan, 'F1-Score': np.nan, 'Support': 0
                        }

                df_metrics = pd.DataFrame.from_dict(data, orient='index')
                print(df_metrics.round(4))

def generate_intra_evaluation_visualizations(intra_results_dict, target_report_labels):
    """
    Generates visualizations for intra-dataset evaluation results.
    Creates bar charts for per-attack metrics for each individual day.

    Args:
        intra_results_dict (dict): The dictionary containing intra-evaluation results.
        target_report_labels (list): List of attack class names to display in specific metrics.
    """
    print("\n--- Generating Intra-Dataset Evaluation Visualizations (Per-Day) ---")

    # Metrics to plot per attack class (Precision, Recall, F1-Score)
    metrics_to_plot_per_attack = ['Precision', 'Recall', 'F1-Score']

    for dataset_name, dataset_results in intra_results_dict.items():
        if isinstance(dataset_results, str) or "Error" in dataset_results:
            print(f"Skipping visualization for {dataset_name} due to error or skip status.")
            continue

        for model_name, model_data in dataset_results.items():
            print(f"\n--- Visualizing Per-Day Metrics for {model_name} in {dataset_name} ---")
            
            per_day_metrics = model_data.get('Per_Day_Test_Metrics', {})

            if not per_day_metrics:
                print(f"No individual daily test results available for {model_name} in {dataset_name} for plotting.")
                continue

            for day_name, day_results in per_day_metrics.items():
                # Skip days that were marked as errors or skipped
                if isinstance(day_results, str) or "Error" in day_results:
                    print(f"  Skipping visualization for day {day_name} due to error or skip status.")
                    continue

                specific_day_metrics = day_results.get('Specific Attack Metrics', {})
                
                plot_data_for_day = []
                for target_label in target_report_labels:
                    if target_label in specific_day_metrics:
                        for metric_type in metrics_to_plot_per_attack:
                            score = specific_day_metrics[target_label].get(metric_type)
                            # Only add if score is not None and not NaN
                            if score is not None and not np.isnan(score):
                                plot_data_for_day.append({
                                    'Attack Type': target_label,
                                    'Metric': metric_type,
                                    'Score': score
                                })

                if plot_data_for_day:
                    df_plot_for_day = pd.DataFrame(plot_data_for_day)
                    
                    # Ensure Attack Type order is consistent for plotting
                    df_plot_for_day['Attack Type'] = pd.Categorical(df_plot_for_day['Attack Type'], categories=target_report_labels, ordered=True)

                    plt.figure(figsize=(10, 6)) # Set figure size for each plot
                    sns.barplot(
                        data=df_plot_for_day,
                        x='Attack Type',
                        y='Score',
                        hue='Metric',
                        palette='viridis'
                    )
                    plt.title(f'Per-Attack Metrics for {model_name} on {dataset_name} - Day: {day_name}')
                    plt.ylim(0, 1.05) # Metrics are usually between 0 and 1
                    plt.xticks(rotation=45, ha='right') # Rotate x-axis labels for readability
                    plt.legend(title='Metric')
                    plt.tight_layout() # Adjust layout to prevent labels from overlapping
                    plt.show()
                else:
                    print(f"  No specific attack metrics found for plotting for day {day_name}.")

def main():
    """
    Main function to orchestrate data loading, preprocessing,
    and cross-dataset/intra-dataset evaluation.
    """
    print("--- Starting Main Evaluation Script ---")
     # --- SCENARIO CONFIGURATION ---
    RUN_FOCUSED_SCENARIO = True # Set to True to focus on specific labels, False for full evaluation
    RUN_CROSS_DATASET_EVALUATION = False # <-- NEW FLAG: Set to False to skip cross-dataset eval

    # NEW: Flags to control retraining for both intra and cross
    FORCE_RETRAIN_INTRA = True # Set to True to force all intra-dataset models to retrain
    FORCE_RETRAIN_CROSS = True # NEW FLAG: Set to True to force all cross-dataset models to retrain

    

    if RUN_FOCUSED_SCENARIO:
        print("\n--- Running Focused Evaluation: Benign, DoS, DDoS, Botnet, Infiltration ---")
        # Define the broad string labels to keep for this focused scenario
        focused_broad_labels_to_keep = ['Benign', 'DoS', 'DDoS',  'Botnet', 'Infiltration']
        
        # Define the specific attack labels to report for this focused scenario
        # This list should only contain the attack classes you want to see in the specific metrics report.
        # 'Benign' is usually handled separately.
        focused_target_report_labels = [ 'Botnet', 'DoS', 'DDoS',  'Infiltration']

        all_combined_dfs, all_individual_dfs_by_dataset, label_encoder, \
        ALL_ENCODED_LABELS, common_features, broad_label_mapper, broad_label_encoder = \
            load_and_align_all_data(DATASET_PATHS, focus_labels=focused_broad_labels_to_keep)
        
        
        # Use the focused report labels for the evaluation functions
        current_target_report_labels = focused_target_report_labels
    else:
        print("\n--- Running Full Multi-Class Evaluation ---")
        all_combined_dfs, all_individual_dfs_by_dataset, label_encoder, \
        ALL_ENCODED_LABELS, common_features, broad_label_mapper, broad_label_encoder = \
            load_and_align_all_data(DATASET_PATHS)
        
        # Use the default target report labels from constants
        current_target_report_labels = TARGET_ATTACK_LABELS_STR_BROAD
    # --- END SCENARIO CONFIGURATION ---

    # --- PLACE THIS DEBUG PRINT HERE ---
    print("\n--- DEBUG (main_evaluation): State of broad_label_encoder after data loading ---")
    print(f"DEBUG (main_evaluation): broad_label_encoder.classes_: {broad_label_encoder.classes_.tolist()}")
    print("-----------------------------------------------------------------------------------\n")

    print("\n--- DEBUG: Types of objects returned from load_and_align_all_data ---")
    print(f"Type of label_encoder: {type(label_encoder)}")
    print(f"Type of ALL_ENCODED_LABELS: {type(ALL_ENCODED_LABELS)}")
    print(f"Type of broad_label_mapper: {type(broad_label_mapper)}")
    print(f"Type of broad_label_encoder: {type(broad_label_encoder)}")
    print("-------------------------------------------------------------------\n")

    # --- END DEBUG PRINT ---

      
    
    # --- Harmonized Label Distribution Check ---
    # This check is performed on the data after granular harmonization and encoding.
    # The labels are decoded back to strings for display purposes.
    print("\n--- Harmonized Label Distribution Check ---")
    
    for dataset_name, df in all_combined_dfs.items():
        print(f"\nLabel distribution for {dataset_name}:")
        
        # Granular Label Distribution (decoded for readability)
        granular_counts_encoded = df['Label'].value_counts()
        granular_labels_decoded = label_encoder.inverse_transform(granular_counts_encoded.index)
        print("Granular Labels (decoded):")
        for i, count in enumerate(granular_counts_encoded):
            print(f"  {granular_labels_decoded[i]}: {count}")

        # Broad Label Distribution (decoded for readability)
        # 'BroadLabel' column is now correctly created and encoded in load_and_align_all_data
        broad_counts_encoded = df['BroadLabel'].value_counts()
        broad_labels_decoded = broad_label_encoder.inverse_transform(broad_counts_encoded.index)

        print("\nBroad Labels (decoded):")

        for i, count in enumerate(broad_counts_encoded):
            print(f"  {broad_labels_decoded[i]}: {count}")

        # Evaluate broad label coverage against TARGET_ATTACK_LABELS_STR_BROAD
        broad_labels_in_df_decoded = set(broad_labels_decoded)
        
        # Ensure TARGET_ATTACK_LABELS_STR_BROAD includes 'Benign' and 'Other Attack' for comparison
        expected_labels_set = set(current_target_report_labels)
        if 'Benign' not in expected_labels_set:
            expected_labels_set.add('Benign')
        if 'Other Attack' not in expected_labels_set:
            expected_labels_set.add('Other Attack')

        unaccounted_labels = broad_labels_in_df_decoded - expected_labels_set
        if unaccounted_labels:
            print(f"  WARNING: Broad labels in {dataset_name} not covered by TARGET_ATTACK_LABELS_STR_BROAD + 'Benign'/'Other Attack': {unaccounted_labels}")

        missing_labels = expected_labels_set - broad_labels_in_df_decoded
        if missing_labels:
            print(f"  INFO: TARGET_ATTACK_LABELS_STR_BROAD includes broad labels not found in {dataset_name}: {missing_labels}")
    print("-------------------------------------------\n")


    # Run Cross-Dataset Evaluation
    cross_results = {}
    if RUN_CROSS_DATASET_EVALUATION and len(all_combined_dfs) >= 2:
        cross_results = run_cross_dataset_evaluation(
            all_combined_dfs,
            all_individual_dfs_by_dataset, # Pass individual dfs for cross-eval per-day testing
            label_encoder,
            ALL_ENCODED_LABELS,
            common_features,
            broad_label_mapper,
            broad_label_encoder,
            current_target_report_labels,
            force_retrain=FORCE_RETRAIN_CROSS
        )
        print("\n--- Cross-Dataset Evaluation Complete ---")

        # NEW: Call the per-day display function for cross-validation
        display_cross_per_day_metrics(cross_results, current_target_report_labels)

        generate_cross_evaluation_visualizations(cross_results, current_target_report_labels)

            
    elif not RUN_CROSS_DATASET_EVALUATION:
        print("Skipping Cross-Dataset Evaluation: Explicitly disabled by configuration.")
    else:
        print("Skipping Cross-Dataset Evaluation: Less than two datasets loaded for cross-evaluation.")





    # Run Intra-Dataset Evaluation (Generalizable Model)
    intra_results = run_intra_dataset_evaluation(
        all_combined_dfs=all_combined_dfs, # Pass combined dfs for overall training/HPO
        all_individual_dfs_by_dataset=all_individual_dfs_by_dataset, # Pass individual dfs for per-day testing
        label_encoder=label_encoder,
        ALL_ENCODED_LABELS=ALL_ENCODED_LABELS,
        common_features=common_features,
        broad_label_mapper=broad_label_mapper,
        broad_label_encoder=broad_label_encoder,
        target_report_labels=current_target_report_labels,
        force_retrain=FORCE_RETRAIN_INTRA
    )
    print("\n--- Intra-Dataset Evaluation Complete ---")

    # --- ADD THIS DEBUG PRINT HERE ---
    print("\n--- DEBUG: Final intra_results dictionary keys and top-level content ---")
    for dataset_key, dataset_value in intra_results.items():
        if isinstance(dataset_value, dict):
            # If it's a dictionary, show its keys (model names)
            print(f"  Dataset: {dataset_key} -> Models: {list(dataset_value.keys())}")
        else:
            # If it's a string (e.g., "Skipped"), print that directly
            print(f"  Dataset: {dataset_key} -> Status: {dataset_value}")
    print("------------------------------------------------------------------\n")
    # --- END DEBUG PRINT ---

     
    display_intra_per_class_metrics(intra_results, current_target_report_labels)

    # NEW: Generate Visualizations
    generate_intra_evaluation_visualizations(intra_results, current_target_report_labels)



if __name__ == "__main__":
    main()
